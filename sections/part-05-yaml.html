  name: my-app

  replicas: 3
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
      - name: app
        image: nginx

Every Kubernetes YAML follows this structure:

  • apiVersion ‚Äì API group
  • kind ‚Äì Resource type
  • metadata ‚Äì Name, labels
